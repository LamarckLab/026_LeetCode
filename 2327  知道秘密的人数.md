#### 2327  知道秘密的人数
---


*01  随便写写*
```python
class Solution(object):
    def peopleAwareOfSecret(self, n, delay, forget):
        """
        :type n: int
        :type delay: int
        :type forget: int
        :rtype: int
        """
        MOD = 10**9 + 7
        dp = [0] * (n + 1)      # dp[i] = 第 i 天新知道秘密的人数
        diff = [0] * (n + 2)    # 差分数组，维护传播人数变化
        
        dp[1] = 1
        diff[1 + delay] += 1    # 第1天知道的人，从 day=1+delay 开始传播
        diff[1 + forget] -= 1   # 从 day=1+forget 开始失效
        
        cur = 0  # 当前传播中的人数
        for day in range(2, n + 1):
            cur = (cur + diff[day]) % MOD
            dp[day] = cur
            # 这些新知道秘密的人会在 [day+delay, day+forget) 之间传播
            if day + delay <= n:
                diff[day + delay] = (diff[day + delay] + dp[day]) % MOD
            if day + forget <= n:
                diff[day + forget] = (diff[day + forget] - dp[day]) % MOD
        
        # 统计还没有忘记秘密的人数（最后 forget-1 天新增的人）
        ans = sum(dp[n - forget + 1 : n + 1]) % MOD
        return ans
```

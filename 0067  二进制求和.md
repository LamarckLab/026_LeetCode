*01 逐位加法o(N)*
```python
class Solution(object):
    def addBinary(self, a, b):
        """
        :type a: str
        :type b: str
        :rtype: str
        """
        i, j = len(a)-1, len(b)-1
        carry = 0
        ans = []
        while i >=0 or j>=0 or carry:
            x = int(a[i]) if i>=0 else 0
            y = int(b[j]) if j>=0 else 0
            digit = (x+y+carry) % 2
            carry = (x+y+carry) // 2
            ans.append(str(digit))
            i -= 1
            j -= 1
        return ''.join(ans[::-1])
```

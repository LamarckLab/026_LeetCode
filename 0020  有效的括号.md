#### 0020 有效的括号
---


*01  栈*
```python
class Solution(object):
    def isValid(self, s):
        """
        :type s: str
        :rtype: bool
        """
        stack = []
        char_dict = {')':'(',']':'[','}':'{'}
        for element in s:
            if element in ['(','[','{']:
                stack.append(element)
            elif element in [')',']','}']:
                if len(stack) < 1:
                    return False
                else:
                    tail = stack.pop()
                    if tail != char_dict[element]:
                        return False
        if len(stack) > 0:
            return False
        elif len(stack) == 0:
            return True
```

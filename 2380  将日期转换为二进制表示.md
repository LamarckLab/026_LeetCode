class Solution(object):
    def convertDateToBinary(self, date):
        """
        :type date: str
        :rtype: str
        """
        date_dcm_list = date.split("-")
        date_bin_list = []
        for element in date_dcm_list:
            bin_element = bin(int(element))[2:]
            date_bin_list.append(bin_element)
        return "-".join(date_bin_list)

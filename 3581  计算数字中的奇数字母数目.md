#### 3581  计算数字中的奇数字母数目
---


*01  随便写写 O(N)*
```python
class Solution(object):
    def countOddLetters(self, n):
        """
        :type n: int
        :rtype: int
        """
        word_dict = {"0":"zero", "1":"one", "2":"two", "3":"three", "4":"four", "5":"five", "6":"six", "7":"seven", "8":"eight", "9":"nine"}
        sentence = ""
        odd_cnt = 0
        for digit in str(n):
            sentence += word_dict[digit]
        frequency_dict = {}
        for letter in sentence:
            if letter not in frequency_dict:
                frequency_dict[letter] = 1
            else:
                frequency_dict[letter] += 1
        for element in frequency_dict:
            if frequency_dict[element] % 2 == 1:
                odd_cnt += 1
        return odd_cnt
```

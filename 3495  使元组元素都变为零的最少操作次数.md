#### 3495  使元组元素都变为零的最少操作次数
---


*01  随便写写*
```python
import bisect
class Solution(object):
    def minOperations(self, queries):
        """
        :type queries: List[List[int]]
        :rtype: int
        """
        ans = 0
        power4 = [1,4,16,64,256,1024,4096,16384,65536,262144,1048576,4194304,16777216,67108864,268435456,1073741824]
        for nums in queries:
            l, r = nums[0], nums[1]
            index = bisect.bisect_left(power4, l)
            operation = 0
            while power4[index] <= r:
               operation += (power4[index] - l) * index
               l = power4[index]
               index += 1
            operation += (r - l + 1) * index
            ans += (operation+1) // 2
        return ans
```

#### 2749  得到整数零需要执行的最少操作数
---


*01  随便写写*
```python
class Solution(object):
    def makeTheIntegerZero(self, num1, num2):
        """
        :type num1: int
        :type num2: int
        :rtype: int
        """
        k = 0
        while k <= 60:
            target = num1 - k*num2
            if target <= 0:
                k += 1
                continue
            else:
                one_cnt = bin(target)[2:].count("1")
                if one_cnt > k or target < k:
                    k += 1
                    continue
                else:
                    return k
        return -1
```

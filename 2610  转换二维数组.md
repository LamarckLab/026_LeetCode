#### 2610  转换二维数组
---


*01  随便写写*
```python
class Solution(object):
    def findMatrix(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        ans_list = []
        num_dict = {}
        sub_list = []
        for digit in nums:
            if digit not in num_dict:
                num_dict[digit] = 1
            else:
                num_dict[digit] += 1
        most_num = max(num_dict, key=num_dict.get)
        most_num_times = num_dict[most_num]
        for i in range(most_num_times):
            for key in num_dict.copy():
                if num_dict[key] > 0:
                    sub_list.append(key)
                    num_dict[key] -= 1
                else:
                    del num_dict[key]
            ans_list.append(sub_list)
            sub_list = []
        return ans_list
```
